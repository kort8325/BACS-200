class Queue:
    def __init__(self):
        self.back = None
        self.queue = []
  
    def enqueue(self,data):
        self.queue.append(data)
        if self.back is None:
            self.back = data
  
    def dequeue(self):
        pop = self.queue[0]
        temp = []
        var = 1
        while(var<=len(self.queue)-1)
            temp.append(self.queue[var])
            var = var +1
        self.queue = temp
        self.back = self.queue[0]
        return pop
  
    def isEmpty(self):
        if self.back is None:
            return True
        else:
            return False
  
    def size(self):
        counter = 0
        for i in self.queue:
            counter = counter + 1
        return counter
