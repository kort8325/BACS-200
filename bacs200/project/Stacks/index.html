Below is code demonstrating python code of a stack being executed. The 'tricks' for this week are included in making the python code

Stack.py
class Stack:
    def __init__(self):
        self.stack = []
        self.top = None
  
    def push(self, data):
        self.stack.append(data)
        self.top = data
  
    def pop(self):
        temp1 = self.stack.pop()
        temp = ""
        for i in self.stack:
            if(i == temp1):
                break
            temp = i
        self.top = temp
        return(temp1)
  
    def peek(self):
        return(self.top)
  
    def isEmpty(self):
        if(self.top is None):
            return True
        return False
  
    def size(self):
        counter = 0
        for i in self.stack:
            counter = counter + 1
        return(counter)
  
# test stack functions above
newStack = Stack()
print(newStack.isEmpty())
newStack.push('a')
newStack.push('b')
newStack.push('c')
newStack.push('d')
print(newStack.peek())
newStack.pop()
print(newStack.stack())
