class Node:

class SLinkedList:
    def __init__(self):
        self.head = None
  
    def add(self, data_in):
        NewNode = Node(data_in)
        NewNode.next = self.head
        self.head = NewNode
  
    def RemoveNode(self, Removekey):

            if (HeadVal.data = Removekey):
                break
            prev = HeadVal.next
        if (HeadVal = None):
            return
        prev.next = HeadVal.next
        HeadVal = None
  
    def search(self, searchKey):
        current = self.head
        while current != None:
            if current.data == serchKey:
                return True
